运行机制

🐍 解释型语言

Python 代码 → 字节码 (.pyc) → Python 虚拟机 (PVM) 逐行执行。没有传统编译过程

🔒 GIL

全局解释器锁（Global Interpreter Lock）。同一时刻只有一个线程执行 Python 字节码。影响 CPU 密集任务，不影响 I/O 密集

⚡ 动态类型

变量没有类型约束，运行时才确定。灵活但可能带来运行时错误。Python 3.6+ 支持类型注解辅助静态检查

🧹 垃圾回收

引用计数为主 + 标记清除（循环引用）。del 减少引用计数，归零立即回收

5.1 类与实例

class Dog:
    """狗类 —— 类文档字符串"""
    species = "Canis familiaris"   # 类属性

    def __init__(self, name, age):   # 构造方法
        self.name = name               # 实例属性
        self.age = age
        self._internal = "受保护"     # 约定：单下划线"保护"
        self.__private = "私有"      # 名称修饰：_Dog__private

        return f"{self.name} says 汪汪!"

    @classmethod
    def create_puppy(cls):           # 类方法（可继承）
        return cls("Puppy", 0)

    @staticmethod
    def info():                      # 静态方法（无 self/cls）
        return "狗是人类的好朋友"

    def __repr__(self):                # 调试用表示
        return f"Dog(name={self.name!r}, age={self.age})"

# 使用
d = Dog("旺财", 3)
print(d.bark())               # 旺财 says 汪汪!
print(Dog.species)            # Canis familiaris
print(Dog.create_puppy())     # 通过类方法创建
print(d._internal)           # 可以访问（但约定不要）

🪄 6. 魔术方法 & 协议

Python 的魔术方法（dunder methods） 让你自定义类的行为——让它像一个内置类型一样工作。

Python
class Vector:
    """实现一个向量 —— 演示常用魔术方法"""
    def __init__(self, x, y):
        self.x, self.y = x, y

    # 字符串表示
    def __repr__(self): return f"Vector({self.x}, {self.y})"
    def __str__(self):  return f"({self.x}, {self.y})"

    # 算术运算
    def __add__(self, other):  return Vector(self.x + other.x, self.y + other.y)
    def __sub__(self, other):  return Vector(self.x - other.x, self.y - other.y)
    def __mul__(self, scalar): return Vector(self.x * scalar, self.y * scalar)
    def __abs__(self):         return (self.x**2 + self.y**2)**0.5

    # 比较
    def __eq__(self, other):   return self.x == other.x and self.y == other.y
    def __lt__(self, other):   return abs(self) < abs(other)

    # 哈希（使对象可用作字典键）
    def __hash__(self): return hash((self.x, self.y))

    # 迭代
    def __iter__(self): return iter((self.x, self.y))

    # 可调用
    def __call__(self): return "向量被调用了！"

    # 上下文管理器
    def __enter__(self): print("进入上下文"); return self
    def __exit__(self, *args): print("退出上下文")

# 测试
v1 = Vector(3, 4)
v2 = Vector(1, 2)
print(v1 + v2)           # (4, 6)
print(v1 * 2)            # (6, 8)
print(abs(v1))           # 5.0
print(v1 > v2)           # True
print(list(v1))          # [3, 4]
print(v1())              # "向量被调用了！"
with v1: print("工作")
🎯 协议速记：
• 可迭代：__iter__ → for x in obj
• 可调用：__call__ → obj()
• 上下文：__enter__ + __exit__ → with obj
• 可哈希：__hash__ + __eq__ → 可用作字典键
• 序列：__len__ + __getitem__ → obj[0]
• 数值：__add__ / __sub__ / __mul__ / ...

8.1 迭代器协议

Python
class CountDown:
    """自定义迭代器：倒计时"""
    def __init__(self, start):
        self.n = start

    def __iter__(self):
        return self              # 迭代器返回自身

    def __next__(self):
        if self.n <= 0:
            raise StopIteration    # 结束信号
        val = self.n
        self.n -= 1
        return val

for x in CountDown(3):
    print(x)                    # 3 2 1

    a, b = 0, 1

🎀 9. 装饰器

装饰器（Decorator） 是 Python 最强大的特性之一——在不修改原函数的情况下增加功能。

9.1 基础装饰器

Python
import time
from functools import wraps

def timer(func):
    """统计函数执行时间"""
    @wraps(func)                         # 保留原函数元信息
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"[{func.__name__}] 耗时: {elapsed:.4f}s")
        return result
    return wrapper

@timer
def slow_function(n):
    return sum(range(n))

slow_function(10_000_000)  # 自动打印执行时间

9.2 带参数的装饰器

Python
def retry(max_attempts=3, delay=1):
    """失败重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    print(f"第 {attempt} 次失败: {e}")
                    if attempt == max_attempts:
                        raise
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unreliable_api():
    import random
    if random.random() < 0.7:
        raise ConnectionError("网络故障")
    return "成功！"

# unreliable_api()

9.3 类装饰器

Python
class Singleton:
    """单例模式装饰器"""
    def __init__(self, cls):
        self._cls = cls
        self._instance = None

    def __call__(self, *args, **kwargs):
        if self._instance is None:
            self._instance = self._cls(*args, **kwargs)
        return self._instance

@Singleton
class Database:
    def __init__(self):
        print("创建数据库连接")

db1 = Database()
db2 = Database()
print(db1 is db2)  # True — 同一个实例

🗑 10. 上下文管理器

上下文管理器 让你用 with 语句自动管理资源——文件、锁、数据库连接，进入时设置，退出时清理。

10.1 类式实现

Python
class ManagedFile:
    def __init__(self, path, mode="r"):
        self.path = path
        self.mode = mode

    def __enter__(self):
        self.file = open(self.path, self.mode, encoding="utf-8")
        print(f"📂 打开文件: {self.path}")
        return self.file       # as 后面拿到的就是这个

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.file.close()
        print(f"📂 关闭文件: {self.path}")
        return False           # False=不吞异常; True=吞掉

14.1 threading（I/O 密集型）

    print(f"[{name}] 开始")
    time.sleep(delay)

    t = threading.Thread(target=worker, args=(f"T-{i}", 2))

    threads.append(t)

    t.join()                          # 等待完成

    global counter
    for _ in range(100000):

            counter += 1

18.1 描述符

class PositiveNumber:
    """描述符：自动校验数值为正"""
    def __set_name__(self, owner, name):
        self.name = f"_{name}"

    def __get__(self, obj, objtype=None):
        return getattr(obj, self.name, 0)

    def __set__(self, obj, value):
        if value <= 0:
            raise ValueError("必须为正数")
        setattr(obj, self.name, value)

class Order:
    amount = PositiveNumber()
    qty = PositiveNumber()

    def __init__(self, amount, qty):
        self.amount = amount   # 触发描述符 __set__
        self.qty = qty

o = Order(100, 2)
print(o.amount)                   # 100

18.2 元类

Python
class SingletonMeta(type):
    """元类实现单例模式"""
    _instances = {}
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Config(metaclass=SingletonMeta):
    pass

a = Config(); b = Config()
print(a is b)  # True

        raise TypeError("只读对象")
