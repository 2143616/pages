// 需求: 将数组按指定大小分块
/* 推理: 需要处理输入验证、
   空数组、负数 size 等边界情况 */
function chunk<T>(arr: T[], size: number): T[][] {
  if (!Array.isArray(arr)) throw new TypeError('arr must be an array');
  if (size < 1) throw new RangeError('size must be >= 1');
  const result: T[][] = [];
  for (let i = 0; i < arr.length; i += size) {
    result.push(arr.slice(i, i + size));
  }
  return result;
}

# 需求: 列表去重并保持顺序（可复用版本）
from typing import TypeVar, List, Hashable

T = TypeVar('T', bound=Hashable)

def dedup(items: List[T]) -> List[T]:
    """去重并保持原始顺序。
    
    Args:
        items: 输入列表（元素需可哈希）
    Returns:
        去重后的列表
    Raises:
        TypeError: 元素不可哈希时
    """
    seen: set[T] = set()
    result: List[T] = []
    for item in items:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result
