1. Linux 是什么

Linux 是一个开源的操作系统内核，广泛用于服务器、嵌入式系统、超级计算机和 AI 训练平台。绝大多数 AI/ML 工具（PyTorch、TensorFlow、CUDA、Docker）都以 Linux 为首要支持平台。

概念说明
Shell命令行解释器，用户通过它和系统交互。最常用的是 Bash。
终端 (Terminal)运行 Shell 的窗口程序。
命令你输入的一行文本，Shell 解析并执行。
文件系统一切皆文件。从根 / 开始的树形结构。
# 打开终端后的第一件事：看看你在哪
whoami        # 当前用户名
pwd           # 当前目录路径
hostname      # 主机名
uname -a      # 系统内核版本信息
💡 学习路径：先掌握文件操作（每天用）→ 系统管理（排查问题）→ 网络工具（部署服务）→ Shell 脚本（自动化）。每一步都有清晰的目标。

📂 文件操作指令

ls — 列出文件和目录

最常用的命令，没有之一。

# 基本用法
ls                    # 列出当前目录（不显示隐藏文件）
ls -l                 # 长格式：权限、大小、修改时间
ls -la                # 包含隐藏文件（以.开头的文件）
ls -lh                # 人类可读的文件大小（K/M/G）
ls -lS                # 按文件大小排序
ls -lt                # 按修改时间排序（最新在最前）
ls -ltr               # 按修改时间反向排序（最旧在最前）
ls -R                 # 递归列出子目录
ls *.py               # 通配符：只列出 .py 文件

输出格式解读（ls -l）：

-rwxr-xr-x 1 hyl hyl 1234 Jul  3 10:00 script.py
^^^^^^^^^ ^   ^^^ ^^^ ^^^^
 权限      链接 用户 组   大小  时间    文件名

权限位 -rwxr-xr-x 解析：第 1 位 -（文件）或 d（目录），后 9 位每 3 位一组：user / group / others，每位 r（读 4）、w（写 2）、x（执行 1）。

cd / pwd — 目录切换

# cd = change directory
cd /home/hyl        # 绝对路径：从根 / 开始
cd Documents        # 相对路径：从当前目录开始
cd ..               # 上一级目录
cd ../..            # 上两级
cd ~                # 家目录（等价于 cd /home/用户名）
cd -                # 上一个目录（来回切换很有用）
cd /                # 根目录

pwd                 # 显示当前目录绝对路径

cp / mv / rm — 复制、移动、删除

# cp = copy
cp file1.txt file2.txt            # 复制文件
cp -r dir1/ dir2/                 # 递归复制目录
cp -i file1.txt dir2/             # 交互模式（覆盖前询问）
cp -v file1.txt dir2/             # 显示复制过程
cp -a dir1/ dir2/                 # 归档模式：保留权限、时间戳

# mv = move（也可用于重命名）
mv old.txt new.txt                # 重命名文件
mv file.txt dir2/                 # 移动文件到目录
mv -i file.txt dir2/              # 覆盖前询问

# rm = remove（⚠️ 没有回收站，删了就没了！）
rm file.txt                       # 删除文件
rm -r dir/                        # 递归删除目录及其内容
rm -rf dir/                       # 强制删除（⚠️ 最危险的命令之一）
rm -i file.txt                    # 删除前确认
⚠️ rm -rf / 会删除整个系统！永远不要运行。重要文件建议用 rm -i 或先 mv 到临时目录。

mkdir / touch — 创建目录和文件

# mkdir = make directory
mkdir new_dir                     # 创建目录
mkdir -p a/b/c/d                  # 递归创建（父目录不存在时自动创建）
mkdir -m 755 dir                  # 创建并设置权限

# touch — 创建空文件 或 更新文件时间戳
touch file.txt                    # 创建空文件（已存在则更新时间戳）
touch -t 202607031000 file.txt    # 设置指定时间戳

find / locate — 查找文件

# find — 实时搜索，功能强大但稍慢
find . -name "*.py"               # 按文件名查找
find . -type f -name "*.txt"      # 只找文件（-type d 找目录）
find . -size +100M                # 查找大于 100MB 的文件
find . -mtime -7                  # 最近 7 天内修改的文件
find . -name "*.log" -delete      # 找到后直接删除（⚠️ 小心）
find . -name "*.py" -exec wc -l {} \;  # 对每个结果执行命令

# locate — 基于数据库，速度极快（需先 updatedb）
locate "*.pdf"                    # 查找所有 PDF
locate -i "README"                # 忽略大小写

👀 文件查看指令

cat / head / tail — 查看文件内容

# cat = concatenate（连接输出）
cat file.txt                      # 显示整个文件
cat -n file.txt                   # 显示行号
cat file1.txt file2.txt > merged.txt # 合并文件

# head — 只看开头
head -n 20 file.txt               # 前 20 行（默认 10 行）
head -c 100 file.txt              # 前 100 个字符

# tail — 只看末尾（调试日志最有用）
tail -n 20 file.txt               # 最后 20 行
tail -f app.log                   # 🔥 实时追踪日志文件（Ctrl+C 退出）
tail -n 100 -f app.log            # 先显示最后 100 行，再实时追踪
💡 tail -f 是调试利器：启动服务时在另一个终端执行 tail -f log.txt，服务日志会实时打印，不用反复打开文件。

less / more — 分页阅读

# less — 功能更强大（推荐）
less large_file.txt               # 分页浏览（空格翻页，q 退出）
less -N file.txt                  # 显示行号
less +F file.txt                  # 类似 tail -f（Ctrl+C 退出追踪模式）
# 浏览中的快捷键：
#   j/k  — 上下滚动一行
#   Ctrl+D/U — 下半页/上半页
#   g/G  — 第一行/最后一行
#   /关键词 — 向前搜索（n 下一个）
#   ?关键词 — 向后搜索
#   q    — 退出

grep — 文本搜索（AI 工程师最常用之一）

# grep = globally search a regular expression
grep "error" app.log              # 搜索包含 error 的行
grep -i "error" app.log           # 忽略大小写
grep -n "error" app.log           # 显示行号
grep -c "error" app.log           # 只计数（多少行匹配）
grep -v "error" app.log           # 反向匹配：不含 error 的行
grep -r "def " ./src/             # 递归搜索目录下所有文件
grep -r --include="*.py" "import" . # 只搜 .py 文件
grep -l "error" *.log             # 只显示匹配的文件名
grep -A3 -B2 "error" log.txt     # 前后文：匹配行前2行+后3行
💡 grep 三部曲：先 grep -n 定位行号 → sed -n '100,120p' 看上下文 → less +120 进入浏览。

wc / sort / uniq — 统计与排序

# wc = word count
wc -l file.txt                    # 行数（最常用）
wc -w file.txt                    # 单词数
wc -c file.txt                    # 字符数（字节数）

# sort — 排序
sort file.txt                     # 字母序排序
sort -n file.txt                  # 按数字排序
sort -r file.txt                  # 反向排序
sort -k2 file.txt                 # 按第二列排序

# uniq — 去重（必须先 sort）
sort file.txt | uniq              # 去重
sort file.txt | uniq -c           # 统计每个元素出现次数
sort file.txt | uniq -d           # 只显示重复的
# 实战：统计日志中每种错误出现次数 top 10
grep "ERROR" app.log | sort | uniq -c | sort -rn | head -10

🔧 系统管理指令

ps / top / htop — 进程管理

# ps = process status
ps aux                            # 🔥 列出所有进程（最常用）
ps aux | grep python              # 只看 Python 进程
ps -ef --forest                   # 树形显示进程父子关系
ps aux --sort=-%mem | head -10   # 按内存使用排序 top10
ps aux --sort=-%cpu | head -10   # 按 CPU 使用排序 top10

# top — 实时进程监视器
top                               # 全屏交互界面（按 q 退出）
# 快捷键：P(按CPU排序) M(按内存) 1(查看多核) k(杀进程)

# htop — top 的增强版（需要安装）
htop                              # 彩色界面，支持鼠标操作

kill — 终止进程

# 先找到进程 PID
ps aux | grep python
# 或用 pgrep
pgrep -l python                   # 列出所有 python 进程的 PID

# 终止进程
kill 12345                        # 优雅终止（SIGTERM）
kill -9 12345                     # 🔥 强制杀死（SIGKILL，慎用）
kill -15 12345                    # 等价于 kill（SIGTERM）

# 批量操作
pkill python                      # 杀死所有 python 进程（⚠️ 小心）
killall python3                   # 同上
⚠️ kill -9 是最后手段：先试普通 kill（SIGTERM），让进程优雅退出（保存数据、释放资源）。只有进程不响应时才用 SIGKILL。

chmod / chown — 权限管理

# chmod = change mode
# 数字方式（推荐）：r=4, w=2, x=1
chmod 755 script.py               # rwxr-xr-x：所有者全权，其他人读+执行
chmod 644 file.txt                # rw-r--r--：所有者读写，其他人只读
chmod 600 secret.txt              # rw-------：仅所有者读写
chmod -R 755 dir/                 # 递归修改目录下所有文件

# 符号方式
chmod +x script.py                # 增加执行权限
chmod u+x script.py               # 仅给所有者增加执行权限
chmod go-w file.txt               # 去掉组和其他人的写权限

# chown = change owner
chown hyl:admins file.txt         # 修改所有者和组
chown -R hyl:admins dir/          # 递归修改

df / du — 磁盘空间管理

# df = disk free — 查看磁盘分区使用情况
df -h                             # 🔥 人类可读格式（最常用）
df -h /                           # 只看根分区
df -h --total                     # 显示总计

# du = disk usage — 查看目录/文件大小
du -sh *                          # 🔥 当前目录下每个项目总大小
du -sh dir/                       # 查看目录总大小
du -h --max-depth=1               # 只看一层子目录的大小
du -sh /home/hly/* | sort -rh    # 按大小排序
# 实战：找出当前目录最大的 5 个文件
du -ah . | sort -rh | head -5

nvidia-smi — GPU 监控（AI 工程师必备）

# 基础信息
nvidia-smi                        # 🔥 GPU 实时状态（显存、温度、进程）

# 查询模式
nvidia-smi --query-gpu=name,memory.used,memory.total,temperature.gpu --format=csv
# 输出：NVIDIA GeForce RTX 2060 SUPER, 7443 MiB, 8192 MiB, 49

# 持续监控
nvidia-smi -l 1                   # 每秒刷新一次
watch -n 1 nvidia-smi             # 用 watch 更稳定

# 查看 GPU 进程
nvidia-smi pmon                   # 进程监控模式
fuser -v /dev/nvidia*             # 查看哪些进程在使用 GPU
💡 watch -n 1 nvidia-smi 是调试 GPU 程序的标配：在另一个终端持续监控显存和利用率，观察训练/推理过程中的 GPU 使用情况。

🌐 网络指令

curl / wget — 网络请求与下载

# curl — 发送 HTTP 请求（API 调试利器）
curl http://localhost:8000/health               # GET 请求
curl -s http://example.com                      # silent 模式（不显示进度）
curl -o output.html http://example.com           # 保存到文件
curl -I https://example.com                      # 只获取响应头

# 🔥 AI 工程师日常：测试推理服务
curl http://localhost:8000/v1/chat/completions \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer token-abc123" \
  -d '{"model":"qwen","messages":[{"role":"user","content":"hi"}]}'
# wget — 下载文件（支持断点续传）
wget https://example.com/file.zip               # 下载文件
wget -c https://example.com/bigfile.zip         # 🔥 断点续传（下载中断后继续）
wget -O output.zip https://example.com/file.zip # 指定保存文件名
wget -i urls.txt                                # 从文件读取多个 URL 批量下载

ssh / scp — 远程连接与文件传输

# SSH 登录远程服务器
ssh hyl@192.168.1.100              # 用户名@IP
ssh -p 2222 hyl@server.com         # 指定端口（默认 22）
ssh -i ~/.ssh/id_rsa hyl@server   # 指定私钥文件

# 🔥 免密登录设置（配一次永久有效）
ssh-keygen -t ed25519             # 生成密钥对（一直回车）
ssh-copy-id hyl@server            # 将公钥复制到服务器
# 之后 ssh 就不再问密码了
# SCP — 安全复制文件
scp file.txt hyl@server:/home/hyl/             # 上传文件到服务器
scp -r dir/ hyl@server:/home/hyl/              # 递归上传目录
scp hyl@server:/home/hyl/file.txt ./            # 从服务器下载

# rsync — 增量同步（🔥 推荐大文件和目录）
rsync -avz dir/ hyl@server:/home/hyl/dir/      # 增量同步
rsync -avz --progress dir/ hyl@server:dir/     # 显示进度
# rsync 只传输差异部分，比 scp 快得多

netstat / ss — 端口与服务查看

# ss — 现代版 netstat（更快）
ss -tlnp                          # 🔥 查看所有监听端口及其进程
ss -tuln                          # TCP + UDP 监听端口
ss -antp | grep 8000              # 查看端口 8000 被谁占用

# 传统 netstat（需安装 net-tools）
netstat -tlnp                     # 查看监听端口
netstat -antp                     # 所有连接
# 实战：端口被占用了怎么办？
ss -tlnp | grep 8000              # 找到占用 8000 端口的进程
kill -9 <PID>                     # 杀死该进程

📦 包管理指令

apt — Debian/Ubuntu 软件包管理

# apt = Advanced Package Tool
sudo apt update                   # 🔥 更新软件包列表（必先执行）
sudo apt upgrade                  # 🔥 升级所有可升级的包
sudo apt install htop             # 安装软件包
sudo apt remove htop              # 卸载
sudo apt autoremove               # 清理无用依赖
sudo apt search "keyword"         # 搜索软件包
sudo apt show htop                # 查看软件包详情
💡 Ubuntu 快捷键：如果你在 vLLM conda 环境里（source activate vllm），sudo 可能找不到。用 sudo env "PATH=$PATH" apt install xxx 或在 root shell 中直接运行。

pip — Python 包管理

# pip = Python Package Installer
pip install numpy                  # 安装包
pip install numpy==1.26.0          # 安装指定版本
pip install -r requirements.txt    # 🔥 批量安装（项目迁移必备）
pip uninstall numpy                # 卸载
pip list                           # 列出已安装包
pip list --outdated               # 列出有更新的包
pip freeze > requirements.txt      # 🔥 导出当前环境依赖

# 国内换源（速度快很多）
pip install numpy -i https://pypi.tuna.tsinghua.edu.cn/simple

# 代理问题（AI 工程师常见）
unset HTTP_PROXY HTTPS_PROXY       # 如果有 SOCKS 冲突，先取消代理
pip install vllm

conda — Python 环境管理（数据科学标配）

# 环境管理
conda create -n myenv python=3.11  # 🔥 创建新环境
conda activate myenv               # 激活环境
conda deactivate                   # 退出环境
conda env list                     # 列出所有环境
conda remove -n myenv --all       # 删除环境

# 包管理
conda install numpy pytorch -c pytorch  # 安装包
conda list                          # 列出当前环境所有包
conda update --all                  # 更新所有包
conda clean --all                   # 清理缓存（释放磁盘空间）

# 环境导出与复制
conda env export > environment.yaml # 🔥 导出环境配置
conda env create -f environment.yaml # 从文件创建环境
💡 conda vs pip：conda 管理整个环境（包括 CUDA、系统库），pip 只管理 Python 包。AI 项目建议先用 conda 建环境，再用 pip 装 Python 包。

⚡ 进阶技巧

管道 | 与重定向 >

这是 Linux 最强大的设计——一个命令的输出作为另一个命令的输入。

# 管道 | — 串联命令
ps aux | grep python               # 在进程列表中搜索 python
cat log.txt | grep "ERROR" | head -5  # 多层管道
dmesg | tail -20                   # 查看内核日志最后 20 行
history | grep "git"               # 搜索历史命令
# 重定向 > 和 >>
echo "hello" > file.txt            # 覆盖写入
echo "world" >> file.txt           # 追加写入
ls -la > output.txt                # 将 ls 的输出保存到文件

# 错误重定向
python script.py 2> error.log      # 只重定向错误（stderr）
python script.py > all.log 2>&1    # 🔥 标准输出和错误都到同一个文件
python script.py &> all.log        # 同上（Bash 4+ 简写）

alias / 环境变量

# alias — 命令别名（省去每次输入长命令）
alias ll='ls -la'                  # 临时设置
alias gpull='git pull origin master'
alias py='python3'

# 🔥 永久别名：写入 ~/.bashrc
echo "alias ll='ls -la'" >> ~/.bashrc
source ~/.bashrc                   # 立即生效
# 环境变量
echo $PATH                         # 查看 PATH（可执行文件搜索路径）
export PATH="/my/bin:$PATH"        # 临时添加路径
export CUDA_VISIBLE_DEVICES=0      # 🔥 指定使用哪块 GPU
export HF_ENDPOINT=https://hf-mirror.com  # HuggingFace 国内镜像
export http_proxy=socks5://127.0.0.1:7897 # 设置代理

# 查看所有环境变量
env
env | grep CUDA                    # 过滤

Shell 脚本基础

#!/bin/bash
# 上面这行叫 shebang，告诉系统用 Bash 解释器

# 变量
name="world"
echo "Hello, $name!"              # 使用变量

# 循环
for i in {1..5}; do
    echo "Count: $i"
done

# 条件
if [ -f "file.txt" ]; then
    echo "文件存在"
fi
# 🔥 实用脚本：批量重命名
for f in *.txt; do
    mv "$f" "${f%.txt}.md"
done
# 运行脚本
chmod +x script.sh                 # 加执行权限
./script.sh                        # 运行
bash script.sh                     # 或用 bash 运行

tmux — 终端会话管理器

在远程服务器上工作必备——即使 SSH 断开，程序也不会中断。

# 安装
sudo apt install tmux

# 基本操作
tmux new -s mysession              # 🔥 创建新会话
tmux detach                        # 断开会话（程序继续运行）
# 快捷键：Ctrl+B 然后 D
tmux attach -t mysession           # 🔥 重新连接会话
tmux ls                            # 列出所有会话

# 分屏（Ctrl+B 后）
#   % — 垂直分屏
#   " — 水平分屏
#   方向键 — 切换分屏
#   Ctrl+方向键 — 调整分屏大小

# 🔥 实战场景：训练模型时
tmux new -s train
# 在 tmux 里运行 python train.py
# Ctrl+B, D 脱离会话
# 即使关闭终端，训练仍在继续
# 回来时：tmux attach -t train

📌 学习建议：
  • 不要死记命令，理解概念和查找方法更重要
  • man 命令 或 命令 --help 随时查看帮助
  • 多练是最好的学习方式——日常操作尽量用命令行
  • 收藏这份文档或打印速查表，随时参考
