💡 1. 数据库基础概念

数据库（Database） 是以一定方式组织、存储和管理数据的仓库。核心目标是持久化、高效查询、并发控制和安全。

关系型 vs 非关系型

特性关系型 (SQL)非关系型 (NoSQL)
数据模型表（行+列），严格 Schema文档/KV/图/列族，灵活 Schema
代表MySQL, PostgreSQL, SQLiteMongoDB, Redis, Neo4j, Cassandra
查询语言SQL（统一标准）各 API 不同
ACID原生支持部分支持（BASE）
适用金融、ERP、电商（强一致）日志、缓存、社交（高并发）

数据库核心术语

🗂 表 (Table)

数据的逻辑容器，由行和列组成。如：users, orders, products

🔑 主键 (Primary Key)

唯一标识一行。不能为 NULL，自动建索引。如：user_id

🔗 外键 (Foreign Key)

引用另一张表的主键，维护数据的参照完整性

📊 索引 (Index)

加速查询的"目录"。类似书的目录页，查询数据更快

📐 视图 (View)

虚拟表，本质是保存的 SQL 查询结果。不存数据

⚡ 事务 (Transaction)

一组操作要么全做，要么全不做。保证数据一致性

🏗 2. SQL — 数据定义 (DDL)

DDL（Data Definition Language）定义数据库结构：创建、修改、删除表。

2.1 CREATE TABLE

SQL
-- 创建用户表
CREATE TABLE users (
    user_id  INTEGER PRIMARY KEY AUTOINCREMENT,
    username TEXT    NOT NULL UNIQUE,
    email    TEXT    NOT NULL UNIQUE,
    age      INTEGER CHECK(age >= 0 AND age <= 150),
    is_active INTEGER DEFAULT 1,
    created_at TEXT  DEFAULT (datetime('now'))
);

-- 创建文章表
CREATE TABLE posts (
    post_id    INTEGER PRIMARY KEY AUTOINCREMENT,
    title      TEXT    NOT NULL,
    content    TEXT    NOT NULL,
    author_id  INTEGER NOT NULL,
    tags       TEXT    DEFAULT '',
    published  INTEGER DEFAULT 0,
    created_at TEXT    DEFAULT (datetime('now')),
    FOREIGN KEY (author_id) REFERENCES users(user_id)
);
Python
def create_tables(conn):
    cursor = conn.cursor()
    cursor.executescript("""
        CREATE TABLE IF NOT EXISTS users (
            user_id  INTEGER PRIMARY KEY AUTOINCREMENT,
            username TEXT    NOT NULL UNIQUE,
            email    TEXT    NOT NULL UNIQUE,
            age      INTEGER DEFAULT 18,
            is_active INTEGER DEFAULT 1,
            created_at TEXT  DEFAULT (datetime('now'))
        );

        CREATE TABLE IF NOT EXISTS posts (
            post_id    INTEGER PRIMARY KEY AUTOINCREMENT,
            title      TEXT    NOT NULL,
            content    TEXT    NOT NULL,
            author_id  INTEGER NOT NULL,
            tags       TEXT    DEFAULT '',
            published  INTEGER DEFAULT 0,
            created_at TEXT    DEFAULT (datetime('now')),
            FOREIGN KEY (author_id) REFERENCES users(user_id)
        );

        CREATE TABLE IF NOT EXISTS comments (
            comment_id INTEGER PRIMARY KEY AUTOINCREMENT,
            post_id    INTEGER NOT NULL,
            author_id  INTEGER NOT NULL,
            body       TEXT    NOT NULL,
            created_at TEXT    DEFAULT (datetime('now')),
            FOREIGN KEY (post_id)   REFERENCES posts(post_id),
            FOREIGN KEY (author_id) REFERENCES users(user_id)
        );
    """)
    conn.commit()
    print("✅ 三张表创建成功")

✏️ 3. SQL — 数据操作 (DML)

DML（Data Manipulation Language）对表中的数据进行增、删、改。

3.1 INSERT

SQL
-- 插入单行
INSERT INTO users (username, email, age)
VALUES ('张三', 'zhang@example.com', 25);

-- 插入多行
INSERT INTO users (username, email, age) VALUES
    ('李四', 'li@example.com', 30),

-- 插入并返回 ID（不同数据库语法不同）
INSERT INTO users (username, email) VALUES ('赵六', 'zhao@test.com');
-- SQLite/MySQL: SELECT last_insert_rowid()
-- PostgreSQL:   INSERT ... RETURNING user_id
Python
def insert_data(conn):
    cursor = conn.cursor()
    users = [
        ("张三", "zhang@example.com", 25),
        ("李四", "li@example.com", 30),
        ("王五", "wang@example.com", 22),
        ("赵六", "zhao@test.com", 28),
        ("孙七", "sun@test.com", 35),
    ]

    cursor.executemany(
        "INSERT INTO users (username, email, age) VALUES (?, ?, ?)",
        users
    )
    conn.commit()
    print(f"✅ 插入了 {cursor.rowcount} 条用户")

    # 插入文章
    posts = [
        ("Python 入门", "Python 是一门优雅的语言...", 1, "python,教程"),
        ("数据库基础", "关系型数据库以表为单位...", 2, "数据库,SQL"),
        ("网络编程", "Socket 是网络编程的基础...", 1, "网络,socket"),
        ("算法设计", "排序算法是计算机科学的基石...", 3, "算法,排序"),
        ("机器学习", "机器学习是人工智能的核心...", 2, "AI,ML"),
    ]
    cursor.executemany(
        "INSERT INTO posts (title, content, author_id, tags) VALUES (?, ?, ?, ?)",
        posts
    )
    conn.commit()
    print(f"✅ 插入了 {cursor.rowcount} 篇文章")

    # 获取最后插入的 ID
    last_id = cursor.lastrowid
    print(f"最后插入的 user_id: {last_id}")

🔍 4. SQL — 数据查询 (DQL)

查询是数据库里最核心、最灵活的操作。一条 SELECT 可以做各种组合。

4.1 基础查询

SQL
-- 查询全部列
SELECT * FROM users;

-- 查询指定列
SELECT username, email, age FROM users;

-- 取别名
SELECT username AS name, email, age FROM users;

-- 去重
SELECT DISTINCT age FROM users;

-- 限制条数
SELECT * FROM users LIMIT 3;
SELECT * FROM users LIMIT 2 OFFSET 1;  -- 跳过1条取2条
Python
def basic_query(conn):
    cursor = conn.cursor()

    cursor.execute("SELECT username, email, age FROM users")
    rows = cursor.fetchall()     # 所有结果
    print("所有用户:")
    for row in rows:
        print(f"  {row[0]:6} | {row[1]:20} | {row[2]:3}岁")

    cursor.execute("SELECT * FROM posts LIMIT 3")
    print("\n前 3 篇文章:")
    for row in cursor.fetchall():
        print(f"  #{row[0]} {row[1]}")

4.2 WHERE 条件

SQL
-- 比较运算符
SELECT * FROM users WHERE age >= 25;
SELECT * FROM users WHERE age BETWEEN 20 AND 30;
SELECT * FROM users WHERE username != '张三';

-- 多条件
SELECT * FROM users WHERE age >= 25 AND is_active = 1;
SELECT * FROM users WHERE age < 20 OR age > 35;

-- IN 操作
SELECT * FROM users WHERE username IN ('张三', '李四', '王五');

-- 模糊匹配 LIKE
SELECT * FROM users WHERE email LIKE '%@example.com';
SELECT * FROM users WHERE username LIKE '张%';  -- 以"张"开头
SELECT * FROM posts WHERE title LIKE '%Python%';  -- 包含 Python

-- NULL 判断
SELECT * FROM users WHERE age IS NULL;
SELECT * FROM users WHERE age IS NOT NULL;
Python
def where_query(conn):
    cursor = conn.cursor()

    # 条件查询
    cursor.execute("SELECT * FROM users WHERE age >= ? AND age <= ?", (25, 35))
    print("年龄 25-35 的用户:")
    for row in cursor.fetchall():
        print(f"  {row[1]} ({row[3]}岁)")

    # LIKE 模糊查询
    cursor.execute("SELECT title FROM posts WHERE title LIKE ?", ("%Python%",))
    print("\n包含 'Python' 的文章:")
    for title in cursor.fetchall():
        print(f"  → {title[0]}")

4.3 排序与分页

SQL
-- 升序 / 降序
SELECT * FROM users ORDER BY age ASC;
SELECT * FROM users ORDER BY age DESC;

-- 多字段排序
SELECT * FROM posts ORDER BY created_at DESC, title ASC;

-- 分页（MySQL/PG 用 LIMIT + OFFSET）
-- 第 1 页 (每页 2 条)
SELECT * FROM posts ORDER BY post_id LIMIT 2 OFFSET 0;
-- 第 2 页
SELECT * FROM posts ORDER BY post_id LIMIT 2 OFFSET 2;
-- 第 N 页: LIMIT page_size OFFSET (page - 1) * page_size
Python
def pagination_query(conn, page=1, page_size=2):
    cursor = conn.cursor()
    offset = (page - 1) * page_size

    cursor.execute(
        "SELECT * FROM posts ORDER BY created_at DESC LIMIT ? OFFSET ?",
        (page_size, offset)
    )
    rows = cursor.fetchall()

    # 总条数
    cursor.execute("SELECT COUNT(*) FROM posts")
    total = cursor.fetchone()[0]

    print(f"第 {page} 页 (共 {total} 条, 每页 {page_size} 条):")
    for row in rows:
        print(f"  #{row[0]} {row[1]} ({row[6]})")

    total_pages = (total + page_size - 1) // page_size
    print(f"  共 {total_pages} 页")

4.4 聚合函数与 GROUP BY

SQL
-- 常用聚合函数
SELECT COUNT(*) FROM users;           -- 总行数
SELECT AVG(age) FROM users;           -- 平均年龄
SELECT MAX(age), MIN(age) FROM users; -- 最大/最小年龄
SELECT SUM(age) FROM users;           -- 年龄总和

-- GROUP BY 分组统计
SELECT age, COUNT(*) AS count
FROM users
GROUP BY age
ORDER BY count DESC;

-- HAVING 过滤分组结果（WHERE 无法过滤聚合结果）
SELECT author_id, COUNT(*) AS post_count
FROM posts
GROUP BY author_id
HAVING post_count >= 2;
Python
def aggregate_query(conn):
    cursor = conn.cursor()

    # 基本聚合
    cursor.execute("SELECT COUNT(*), AVG(age), MAX(age), MIN(age) FROM users")
    count, avg, mx, mn = cursor.fetchone()
    print(f"用户统计: {count}人, 平均{avg:.1f}岁, 最大{mx}岁, 最小{mn}岁")

    # 按年龄分组
    cursor.execute("SELECT age, COUNT(*) FROM users GROUP BY age ORDER BY age")
    print("\n年龄分布:")
    for age, cnt in cursor.fetchall():
        bar = "█" * cnt
        print(f"  {age:>2}岁: {bar} ({cnt}人)")

    # 统计每位作者的文章数
    cursor.execute("""
        SELECT u.username, COUNT(p.post_id) AS post_count
        FROM users u
        LEFT JOIN posts p ON u.user_id = p.author_id
        GROUP BY u.user_id
        ORDER BY post_count DESC
    """)
    print("\n文章数量统计:")
    for name, cnt in cursor.fetchall():
        print(f"  {name}: {cnt} 篇")

🧩 5. 高级查询 & 子查询

5.1 子查询

SQL
-- 标量子查询（返回单值）
SELECT username,
       (SELECT COUNT(*) FROM posts WHERE author_id = users.user_id) AS post_count
FROM users;

-- 行子查询（返回单行）
SELECT * FROM users
WHERE (age, username) = (SELECT MAX(age), username FROM users);

-- 表子查询（IN）
SELECT * FROM posts
WHERE author_id IN (SELECT user_id FROM users WHERE age > 25);

-- EXISTS 检查存在性
SELECT * FROM users u
WHERE EXISTS (SELECT 1 FROM posts p WHERE p.author_id = u.user_id);
Python
def subquery_demo(conn):
    cursor = conn.cursor()

    -- 每个用户的文章数（标量子查询）
    cursor.execute("""
        SELECT username, email,
               (SELECT COUNT(*) FROM posts WHERE author_id = users.user_id) AS post_count
        FROM users
        ORDER BY post_count DESC
    """)
    print("用户文章数（子查询）:")
    for name, email, cnt in cursor.fetchall():
        print(f"  {name:6} | {email:20} | {cnt} 篇")

    -- 找写过文章的用户（EXISTS）
    cursor.execute("""
        SELECT user_id, username FROM users u
        WHERE EXISTS (SELECT 1 FROM posts p WHERE p.author_id = u.user_id)
    """)
    print("\n发过文章的用户:")
    for uid, name in cursor.fetchall():
        print(f"  {name} (ID:{uid})")

5.2 窗口函数

SQL
-- 窗口函数：为每行计算一个基于"窗口"的值

-- ROW_NUMBER: 按年龄排序编号
SELECT username, age,
       ROW_NUMBER() OVER (ORDER BY age DESC) AS rank
FROM users;

-- RANK / DENSE_RANK: 排名（考虑并列）
SELECT username, age,
       RANK()       OVER (ORDER BY age DESC) AS r1,
       DENSE_RANK() OVER (ORDER BY age DESC) AS r2
FROM users;

-- SUM 窗口函数：累计求和
SELECT post_id, title, author_id,
       COUNT(*) OVER (PARTITION BY author_id) AS author_total
FROM posts;
Python
def window_function_demo(conn):
    cursor = conn.cursor()
    # SQLite 3.25+ 支持窗口函数
    cursor.execute("""
        SELECT username, age,
               ROW_NUMBER() OVER (ORDER BY age DESC) AS rn,
               DENSE_RANK() OVER (ORDER BY age DESC) AS dr
        FROM users
        ORDER BY age DESC
    """)
    print("年龄排行（窗口函数）:")
    print("  {:8} {:4} {:7} {:7}".format("姓名", "年龄", "行号", "排名"))
    for name, age, rn, dr in cursor.fetchall():
        print(f"  {name:8} {age:3} {rn:7} {dr:7}")

🔗 6. 表连接 (JOIN)

关系型数据库的"关系"二字核心就在 JOIN——把多张表按关联字段连接起来。

6.1 INNER JOIN

SQL
-- 文章 + 作者（只返回匹配上的）
SELECT p.title, u.username
FROM posts p
INNER JOIN users u ON p.author_id = u.user_id;

-- 三表连接：文章 + 作者 + 评论
SELECT p.title, u.username AS author, c.body AS comment
FROM posts p
JOIN users u    ON p.author_id = u.user_id
JOIN comments c ON c.post_id = p.post_id;

6.2 LEFT / RIGHT / FULL JOIN

SQL
-- LEFT JOIN: 左表全部保留，右表无匹配则 NULL
SELECT u.username, p.title
FROM users u
LEFT JOIN posts p ON u.user_id = p.author_id;
-- 结果：所有用户都会出现，没写文章的 title 为 NULL

-- 找出没写文章的用户
SELECT u.username
FROM users u
LEFT JOIN posts p ON u.user_id = p.author_id
WHERE p.post_id IS NULL;

-- FULL OUTER JOIN: 两张表全部保留（SQLite 不直接支持）
-- MySQL 不直接支持 FULL JOIN，用 UNION 模拟

6.3 自连接

SQL
-- 创建部门表演示自连接
CREATE TABLE employees (
    emp_id   INTEGER PRIMARY KEY,
    name     TEXT,
    manager_id INTEGER REFERENCES employees(emp_id)
);

-- 查询员工 + 上级领导
SELECT e.name AS employee, m.name AS manager
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.emp_id;

6.4 JOIN 综合示例

Python
def join_demo(conn):
    cursor = conn.cursor()

    # 文章 + 作者名称
    cursor.execute("""
        SELECT p.title, u.username, p.created_at
        FROM posts p
        JOIN users u ON p.author_id = u.user_id
        ORDER BY p.created_at DESC
    """)
    print("文章与作者:")
    for title, author, dt in cursor.fetchall():
        print(f"  📝 {title:12} | ✍ {author:6} | {dt}")

    # LEFT JOIN：所有用户及其文章
    cursor.execute("""
        SELECT u.username, COALESCE(p.title, '(未写文章)')
        FROM users u
        LEFT JOIN posts p ON u.user_id = p.author_id
        ORDER BY u.user_id
    """)
    print("\n所有用户与文章:")
    for name, title in cursor.fetchall():
        print(f"  {name:6} → {title}")
💡 JOIN 记忆口诀：
• INNER JOIN = 取交集，两边都有才返回
• LEFT JOIN = 左表全保留，右表无匹配给 NULL
• RIGHT JOIN = 右表全保留（MySQL 支持，SQLite 不支持）
• FULL JOIN = 两表并集（两边全保留）
• CROSS JOIN = 笛卡尔积（每行 × 每行，很少用）

📑 7. 索引

索引（Index） 是数据库高效查询的核心机制——以空间换时间，类似书的目录。

7.1 索引类型

索引类型底层结构适用场景
B+ Tree 索引平衡多路搜索树大多数场景（等值/范围/排序）
哈希索引哈希表等值查询（不支持范围）
全文索引倒排索引LIKE / 文本搜索
空间索引R-Tree地理位置查询
位图索引位图低基数列（性别/状态）

7.2 索引操作

SQL
-- 创建单列索引
CREATE INDEX idx_users_email ON users(email);

-- 创建唯一索引
CREATE UNIQUE INDEX idx_users_username ON users(username);

-- 创建复合索引（多列顺序重要）
CREATE INDEX idx_posts_author_date ON posts(author_id, created_at DESC);
-- 查询 WHERE author_id = ? ORDER BY created_at DESC => 走此索引

-- 删除索引
DROP INDEX IF EXISTS idx_users_email;

-- EXPLAIN 查看查询是否走索引
EXPLAIN QUERY PLAN SELECT * FROM users WHERE email = 'test@test.com';
Python
def index_demo(conn):
    cursor = conn.cursor()

    # 创建索引
    cursor.executescript("""
        CREATE INDEX IF NOT EXISTS idx_users_email ON users(email);
        CREATE INDEX IF NOT EXISTS idx_posts_author ON posts(author_id);
        CREATE INDEX IF NOT EXISTS idx_posts_created ON posts(created_at);
    """)
    conn.commit()
    print("✅ 索引创建完成")

    # 查看索引列表
    cursor.execute("SELECT name, tbl_name FROM sqlite_master WHERE type = 'index' AND name NOT LIKE 'sqlite_%'")
    print("\n数据库中的索引:")
    for idx_name, tbl in cursor.fetchall():
        print(f"  📌 {idx_name:25} → {tbl}")

    # EXPLAIN QUERY PLAN：分析查询是否使用索引
    cursor.execute("EXPLAIN QUERY PLAN SELECT * FROM users WHERE email = 'test@test.com'")
    print("\n查询计划:")
    for row in cursor.fetchall():
        print("  ", row)

7.3 索引注意事项

✅ 适合建索引

WHERE 频繁的列、JOIN 关联的列、ORDER BY 的列、唯一约束列

❌ 不要建索引

小表（<100行）、频繁更新/插入的表、值极少变化的列（如性别）

📏 复合索引最左前缀

索引 (a, b, c) -> WHERE a=1 走索引; WHERE b=2 不走; WHERE a=1 AND b=2 走

🔄 维护成本

索引会降低 INSERT/UPDATE/DELETE 速度（因为要同时更新索引树）

Python
def explain_comparison(conn):
    cursor = conn.cursor()

    # 不加索引的查询
    cursor.execute("EXPLAIN QUERY PLAN SELECT * FROM users WHERE age = 25")
    print("无索引的查询计划:")
    for row in cursor.fetchall():
        detail = row[3] if len(row) > 3 else row
        print(f"  → {detail} (全表扫描)")

    # 加索引后
    cursor.execute("CREATE INDEX IF NOT EXISTS idx_users_age ON users(age)")
    cursor.execute("EXPLAIN QUERY PLAN SELECT * FROM users WHERE age = 25")
    print("\n有索引的查询计划:")
    for row in cursor.fetchall():
        detail = row[3] if len(row) > 3 else row

8.2 三大范式

范式规则违反示例
1NF (第一范式)每列不可再分（原子性）tags 列存 "Python,SQL,数据库" → 违反
2NF (第二范式)满足 1NF，且非主键列完全依赖于主键订单表: (order_id, product_id, product_name, qty) → product_name 只依赖 product_id，不是全部主键
3NF (第三范式)满足 2NF，且非主键列不传递依赖于主键员工表: (emp_id, dept_id, dept_name) → dept_name 通过 dept_id 传递依赖于 emp_id

⚖️ 9. 事务与 ACID

9.1 ACID 特性

A — 原子性

事务里的操作要么全做，要么全不做。如转账：A扣钱+B加钱，有一条失败就回滚

C — 一致性

事务前后数据满足所有约束（主键、外键、CHECK）。不违反业务规则

I — 隔离性

并发执行的事务互不干扰。通过隔离级别控制

D — 持久性

事务一旦提交，数据就永久保存（写入磁盘）

9.2 隔离级别

隔离级别脏读不可重复读幻读

def transaction_demo(conn):
    cursor = conn.cursor()

    print("=== 事务演示：银行转账 ===\n")

    # 建表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS accounts (
            acc_id INTEGER PRIMARY KEY,
            name    TEXT    NOT NULL,
            balance REAL    NOT NULL DEFAULT 0
        )
    """)
    cursor.execute("DELETE FROM accounts")
    cursor.execute("INSERT INTO accounts VALUES (1, 'Alice', 1000)")
    cursor.execute("INSERT INTO accounts VALUES (2, 'Bob', 500)")
    conn.commit()

    def show_balance():
        cursor.execute("SELECT * FROM accounts")
        for acc_id, name, bal in cursor.fetchall():
            print(f"  {name:8} 余额: ${bal:>7.2f}")

    print("转账前余额:")
    show_balance()

    # 事务开始：Alice 转 $200 给 Bob
    try:
        conn.execute("BEGIN TRANSACTION")

        # 1. Alice 扣钱
        cursor.execute(
            "UPDATE accounts SET balance = balance - 200 WHERE acc_id = 1"
        )
        # 检查 Alice 余额是否足够
        cursor.execute("SELECT balance FROM accounts WHERE acc_id = 1")
        if cursor.fetchone()[0] < 0:
            raise ValueError("余额不足！")

        # 2. Bob 加钱
        cursor.execute(
            "UPDATE accounts SET balance = balance + 200 WHERE acc_id = 2"
        )

        # 3. 提交
        conn.commit()
        print("\n✅ 转账成功! Alice → Bob $200")

    except Exception as e:
        conn.rollback()
        print(f"\n❌ 转账失败! 回滚事务: {e}")

    print("\n转账后余额:")
    show_balance()

import threading
import time

def deadlock_demo():
    """
    模拟数据库死锁：
    事务1: UPDATE A → UPDATE B
    事务2: UPDATE B → UPDATE A
    """
    db = sqlite3.connect(":memory:")
    db.execute("CREATE TABLE t (id INT PRIMARY KEY, val INT)")
    db.execute("INSERT INTO t VALUES (1, 10), (2, 20)")
    db.commit()
    db.execute("PRAGMA journal_mode=WAL")

    results = []

    def tx1():
        conn = sqlite3.connect(":memory:")
        conn.execute("INSERT INTO t VALUES (1, 10), (2, 20)")
        conn.commit()
        try:
            conn.execute("BEGIN IMMEDIATE")
            conn.execute("UPDATE t SET val = val * 2 WHERE id = 1")
            time.sleep(0.05)
            conn.execute("UPDATE t SET val = val * 2 WHERE id = 2")
            conn.commit()
            results.append("tx1 OK")
        except Exception as e:
            results.append(f"tx1 ERROR: {e}")
        finally:
            conn.close()

    def tx2():
        conn = sqlite3.connect(":memory:")
        conn.execute("INSERT INTO t VALUES (1, 10), (2, 20)")
        conn.commit()
        try:
            conn.execute("BEGIN IMMEDIATE")
            conn.execute("UPDATE t SET val = val + 5 WHERE id = 2")
            time.sleep(0.05)
            conn.execute("UPDATE t SET val = val + 5 WHERE id = 1")
            conn.commit()
            results.append("tx2 OK")
        except Exception as e:
            results.append(f"tx2 ERROR: {e}")
        finally:
            conn.close()

    t1 = threading.Thread(target=tx1)
    t2 = threading.Thread(target=tx2)
    t1.start(); t2.start()
    t1.join(); t2.join()

    print("死锁测试结果:")
    for r in results:
        print("  ", r)
    print("✅ 解决：按固定顺序获取资源 或 设置超时重试")

def savepoint_demo(conn):
    """Savepoint: 事务内的子事务，可以部分回滚"""
    cursor = conn.cursor()

    try:
        cursor.execute("BEGIN")

        cursor.execute("INSERT INTO users (username, email) VALUES ('A', 'a@t.com')")

        # 设置保存点
        cursor.execute("SAVEPOINT sp1")
        cursor.execute("INSERT INTO users (username, email) VALUES ('B', 'b@t.com')")
        cursor.execute("INSERT INTO users (username, email) VALUES ('C', 'c@t.com')")

        # 模拟出错，回滚到保存点
        print("模拟插入 C 出错，回滚到 savepoint...")
        cursor.execute("ROLLBACK TO SAVEPOINT sp1")

        # 释放保存点
        cursor.execute("RELEASE SAVEPOINT sp1")

        conn.commit()
        print("✅ 提交成功 (B和C被回滚，只有A插入)")

    except Exception as e:
        conn.rollback()
        print("❌ 回滚:", e)

13.1 完整的数据导入导出工具

import csv
import json
import io
from typing import List, Dict

class DatabaseToolkit:
    """数据库工具包 —— 导入、导出、报表"""

    def __init__(self, db_path="app.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row

    def export_to_csv(self, table: str, output_file: str):
        cursor = self.conn.execute(f"SELECT * FROM {table}")
        rows = cursor.fetchall()
        if not rows: return

        columns = [d[0] for d in cursor.description]
        with open(output_file, "w", newline="", encoding="utf-8") as f:
            writer = csv.writer(f)
            writer.writerow(columns)
            writer.writerows([[row[c] for c in columns] for row in rows])
        print(f"✅ 导出 {table} → {output_file} ({len(rows)} 行)")

    def export_to_json(self, table: str, output_file: str):
        cursor = self.conn.execute(f"SELECT * FROM {table}")
        rows = [dict(row) for row in cursor.fetchall()]
        # JSON 序列化 datetime 等
        def serialize(v):
            if isinstance(v, (datetime,)):
                return v.isoformat()
            return v
        for row in rows:
            for k in row:
                row[k] = serialize(row[k])

        with open(output_file, "w", encoding="utf-8") as f:
            json.dump(rows, f, ensure_ascii=False, indent=2)
        print(f"✅ 导出 {table} → {output_file} ({len(rows)} 行)")

    def import_from_csv(self, table: str, csv_file: str):
        with open(csv_file, encoding="utf-8") as f:
            reader = csv.DictReader(f)
            rows = [row for row in reader]

        if not rows: return
        columns = list(rows[0].keys())
        placeholders = ", ".join(["?"] * len(columns))
        sql = f"INSERT INTO {table} ({', '.join(columns)}) VALUES ({placeholders})"
        values = [[row[c] for c in columns] for row in rows]
        self.conn.executemany(sql, values)
        self.conn.commit()
        print(f"✅ 从 {csv_file} 导入 {table} ({len(rows)} 行)")

    def generate_report(self):
        """生成数据库统计报告"""
        cursor = self.conn.cursor()
        report = {}

        cursor.execute("SELECT name FROM sqlite_master WHERE type='table' ORDER BY name")
        tables = [row[0] for row in cursor.fetchall()]

        for table in tables:
            cursor.execute(f"SELECT COUNT(*) FROM {table}")
            row_count = cursor.fetchone()[0]
            cursor.execute(f"PRAGMA table_info({table})")
            col_count = len(cursor.fetchall())
            report[table] = {"rows": row_count, "columns": col_count}

        print("\n📊 数据库统计报告")
        print("=" * 50)
        total_rows = 0
        for table, info in report.items():
            print(f"  {table:15} | {info['columns']:>3} 列 | {info['rows']:>6} 行")
            total_rows += info["rows"]
        print("=" * 50)
        print(f"  共 {len(tables)} 张表, {total_rows} 行数据")
        return report

    def close(self):
        self.conn.close()

if __name__ == __name__:
    print("=== 数据库完整演示 ===\n")

    # 1. 创建数据库 + 表
    conn = sqlite3.connect(":memory:")
    cur = conn.cursor()
    cur.executescript("""
        CREATE TABLE users(
            user_id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            email TEXT UNIQUE,
            age INTEGER DEFAULT 18
        );
        CREATE TABLE orders(
            order_id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id INTEGER REFERENCES users(user_id),
            product TEXT NOT NULL,
            price REAL NOT NULL,
            qty INTEGER DEFAULT 1,
            order_date TEXT DEFAULT (datetime('now'))
        );
    """)

    # 2. 插入数据
    cur.executemany("INSERT INTO users(name, email, age) VALUES (?,?,?)",
        [("Alice", "alice@t.com", 25),
         ("Bob", "bob@t.com", 32),
         ("Charlie", "charlie@t.com", 28)])
    cur.executemany("INSERT INTO orders(user_id, product, price, qty) VALUES (?,?,?,?)",
        [(1, "笔记本电脑", 5999, 1),
         (1, "鼠标", 99, 2),
         (2, "键盘", 299, 1),
         (3, "显示器", 1999, 1)])
    conn.commit()

    # 3. 查询：用户消费总额
    cur.execute("""
        SELECT u.name,
               COUNT(o.order_id) AS orders,
               ROUND(SUM(o.price * o.qty), 2) AS total_spent
        FROM users u
        LEFT JOIN orders o ON u.user_id = o.user_id
        GROUP BY u.user_id
        ORDER BY total_spent DESC
    """)
    print("用户消费统计:")
    for name, cnt, total in cur.fetchall():
        print(f"  {name:10} | 订单 {cnt} 笔 | 总共 ¥{total:>8.2f}")

    # 4. 找出最贵订单
    cur.execute("""
        SELECT u.name, o.product, o.price * o.qty AS total
        FROM orders o JOIN users u ON o.user_id = u.user_id
        ORDER BY total DESC LIMIT 1
    """)
    name, prod, total = cur.fetchone()
    print(f"\n🏆 最贵订单: {name} 购买了 {prod} (¥{total})")

    # 5. 使用窗口函数排名
    cur.execute("""
        SELECT name, age,
               ROW_NUMBER() OVER (ORDER BY age DESC) AS rank
        FROM users
    """)
    print("\n用户年龄排名:")
    for name, age, rn in cur.fetchall():
        print(f"  #{rn} {name:10} {age}岁")

    conn.close()
    print("\n✅ 完整演示结束!")

13.3 SQL 面试高频题速查

1. 第二高薪水

SELECT MAX(salary) FROM emp WHERE salary < (SELECT MAX(salary) FROM emp)

2. 部门最高薪

SELECT dept, MAX(salary) FROM emp GROUP BY dept

3. 连续登录 N 天

DATE(login_date) - ROW_NUMBER() OVER (PARTITION BY uid ORDER BY date) 分组

4. 查重复邮箱

SELECT email, COUNT(*) FROM users GROUP BY email HAVING COUNT(*) > 1

5. 删除重复行

DELETE FROM users WHERE rowid NOT IN (SELECT MIN(rowid) FROM users GROUP BY email)

6. 累加求和

SELECT date, amount, SUM(amount) OVER (ORDER BY date) AS cumulative FROM sales
