🧭 1. 数据结构总览

数据结构 是计算机存储、组织数据的方式。选择合适的数据结构可以极大地影响算法效率——一个好结构能让 O(n²) 变成 O(log n)。

两大维度

📦 线性结构

数据元素排列成一条线：数组、链表、栈、队列。遍历方式单一（前→后）

🌳 非线性结构

数据元素之间有层次/网状关系：树、图、集合。遍历方式多样（DFS / BFS）

数据结构 vs. 算法

名言
程序 = 数据结构 + 算法
           — Niklaus Wirth

数据结构决定「怎么存」，算法决定「怎么算」。
选错了结构，再好的算法也救不了。

例：
  — 需要频繁按索引访问? → 数组 O(1)
  — 需要频繁在中间插入? → 链表 O(1)
  — 需要快速查找?        → 哈希表 O(1) / BST O(log n)
场景推荐结构原因
表达式求值栈后进先出天然匹配括号/运算
任务排队队列先来先服务
路由器转发字典树 (Trie)IP 前缀匹配 O(len)
社交关系推荐图朋友关系 = 图的边
寻找最短路径图 + Dijkstra加权最短路径
稀疏数据查找哈希表平均 O(1) 查找

手写动态数组（理解底层）

Python
class DynamicArray:
    """手动实现动态数组（模拟 Python list 底层）"""
    def __init__(self, capacity=2):
        self._capacity = capacity
        self._size = 0
        self._data = [None] * capacity

    def __len__(self): return self._size

    def __getitem__(self, i):
        if i < 0 or i >= self._size:
            raise IndexError("out of range")
        return self._data[i]

    def append(self, val):
        if self._size == self._capacity:
            self._resize(self._capacity * 2)  # 翻倍扩容
        self._data[self._size] = val
        self._size += 1

    def pop(self):
        if self._size == 0:
            raise IndexError("pop from empty")
        val = self._data[self._size - 1]
        self._size -= 1
        if self._size > 0 and self._size <= self._capacity // 4:
            self._resize(self._capacity // 2)  # 缩容
        return val

    def _resize(self, new_cap):
        new_data = [None] * new_cap
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data = new_data
        self._capacity = new_cap

    def __repr__(self):
        return "[" + ", ".join(str(self._data[i]) for i in range(self._size)) + "]"

# 测试
da = DynamicArray()
for i in range(8): da.append(i)
print(da, "容量:", da._capacity)  # [0, 1, 2, 3, 4, 5, 6, 7] 容量: 8

[[0] * 5] * 3 创建的不是 3 行 5 列——3 个引用指向同一个列表！应用列表推导式：[[0] * 5 for _ in range(3)]

⛓ 3. 链表

链表（Linked List） 是由节点组成的线性结构，每个节点包含数据 + 指向下一个节点的指针。不同于数组的连续内存，链表的节点在内存中离散分布。

3.1 单向链表

Python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, val):          # 尾部添加
        if not self.head:
            self.head = ListNode(val)
            return
        cur = self.head
        while cur.next: cur = cur.next
        cur.next = ListNode(val)

    def prepend(self, val):         # 头部添加
        self.head = ListNode(val, self.head)

    def delete(self, val):          # 删除第一个匹配值
        if not self.head: return
        if self.head.val == val:
            self.head = self.head.next
            return
        cur = self.head
        while cur.next and cur.next.val != val:
            cur = cur.next
        if cur.next:
            cur.next = cur.next.next

    def find(self, val) -> int:   # 查找 → 索引
        cur, i = self.head, 0
        while cur:
            if cur.val == val: return i
            cur = cur.next; i += 1
        return -1

    def to_list(self) -> list:
        result, cur = [], self.head
        while cur:
            result.append(cur.val)
            cur = cur.next
        return result

# 测试
ll = LinkedList()
ll.append(1); ll.append(2); ll.append(3)
ll.prepend(0)
print(ll.to_list())  # [0, 1, 2, 3]
ll.delete(2)
print(ll.to_list())  # [0, 1, 3]

3.2 反转链表（高频面试题）

Python
def reverse_list(head: ListNode) -> ListNode:
    """反转单向链表（迭代法）"""
    prev = None
    cur = head
    while cur:
        next_node = cur.next   # 保存下一个
        cur.next = prev        # 指向前一个
        prev = cur             # prev 前进
        cur = next_node        # cur 前进
    return prev               # 新的头

def reverse_recursive(head: ListNode) -> ListNode:
    """反转链表（递归法，更优雅）"""
    if not head or not head.next:
        return head
    new_head = reverse_recursive(head.next)
    head.next.next = head       # 让下一个节点指向自己
    head.next = None
    return new_head

# 测试
ll = LinkedList()
for i in range(5): ll.append(i)
print("原:", ll.to_list())
ll.head = reverse_recursive(ll.head)
print("反转后:", ll.to_list())

3.3 双向链表

Python
class DListNode:
    def __init__(self, val=0, prev=None, next=None):
        self.val = val
        self.prev = prev
        self.next = next

class DoublyLinkedList:
    """双向链表（带哨兵头尾）"""
    def __init__(self):
        self.head = DListNode()       # 哨兵
        self.tail = DListNode()
        self.head.next = self.tail
        self.tail.prev = self.head
        self._size = 0

    def __len__(self): return self._size

    def append(self, val):
        node = DListNode(val)
        node.prev = self.tail.prev
        node.next = self.tail
        self.tail.prev.next = node
        self.tail.prev = node
        self._size += 1

    def prepend(self, val):
        node = DListNode(val)
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node
        self._size += 1

    def pop_left(self):
        if self._size == 0: raise IndexError
        node = self.head.next
        val = node.val
        self.head.next = node.next
        node.next.prev = self.head
        self._size -= 1
        return val

    def pop_right(self):
        if self._size == 0: raise IndexError
        node = self.tail.prev
        val = node.val
        self.tail.prev = node.prev
        node.prev.next = self.tail
        self._size -= 1
        return val

    def to_list(self):
        result = []
        cur = self.head.next
        while cur is not self.tail:
            result.append(cur.val)
            cur = cur.next
        return result

3.4 快慢指针典型题

Python
def has_cycle(head: ListNode) -> bool:
    """检测链表是否有环（Floyd 判圈算法）"""
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow is fast:
            return True
    return False

def find_middle(head: ListNode) -> ListNode:
    """找链表中间节点"""
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow

def get_intersection(headA: ListNode, headB: ListNode) -> ListNode | None:
    """找出两个链表的交点（相交链表）"""
    a, b = headA, headB
    while a is not b:
        a = a.next if a else headB
        b = b.next if b else headA
    return a

📚 4. 栈

栈（Stack） 是后进先出（LIFO）的线性结构。只能在一端（栈顶）进行添加和删除——就像一叠盘子。

4.1 栈的实现

# 手写实现（链表栈）
class Stack:
    def __init__(self):
        self._items = []

    def push(self, val):  self._items.append(val)
    def pop(self):        return self._items.pop()
    def peek(self):       return self._items[-1]
    def is_empty(self):   return len(self._items) == 0
    def __len__(self):     return len(self._items)

4.2 括号匹配

Python
def is_valid_brackets(s: str) -> bool:
    """判断括号是否有效匹配"""
    pairs = {")": "(", "]": "[", "}": "{"}
    stack = []
    for ch in s:
        if ch in pairs:
            if not stack or stack.pop() != pairs[ch]:
                return False
        else:
            stack.append(ch)
    return not stack

print(is_valid_brackets("({[]})"))  # True
print(is_valid_brackets("([)]"))    # False

4.3 逆波兰表达式计算

Python
def eval_rpn(tokens: list[str]) -> int:
    """计算逆波兰表达式（后缀表达式）"""
    stack = []
    operators = {"+", "-", "*", "/"}
    for t in tokens:
        if t in operators:
            b, a = stack.pop(), stack.pop()
            match t:
                case "+": stack.append(a + b)
                case "-": stack.append(a - b)
                case "*": stack.append(a * b)
                case "/": stack.append(int(a / b))  # 向零取整
        else:
            stack.append(int(t))
    return stack[0]

print(eval_rpn(["2", "1", "+", "3", "*"]))  # (2+1)*3 = 9
print(eval_rpn(["4", "13", "5", "/", "+"]))    # 4 + 13/5 = 6

4.4 单调栈

Python
def next_greater_element(nums: list[int]) -> list[int]:
    """下一个更大元素（单调递减栈）"""
    n = len(nums)
    result = [-1] * n
    stack = []                       # 存索引（单调减）
    for i in range(n):
        while stack and nums[i] > nums[stack[-1]]:
            idx = stack.pop()
            result[idx] = nums[i]
        stack.append(i)
    return result

print(next_greater_element([2, 1, 5, 3, 4]))
# [5, 5, -1, 4, -1]

class Queue:
    def __init__(self):
        self._items = deque()

    def enqueue(self, val):   self._items.append(val)
    def dequeue(self):       return self._items.popleft()
    def front(self):         return self._items[0]
    def is_empty(self):      return len(self._items) == 0
    def __len__(self):        return len(self._items)

5.3 循环队列（数组实现）

Python
class CircularQueue:
    """循环队列 — 用固定数组实现队列"""
    def __init__(self, capacity=5):
        self._data = [None] * capacity
        self._cap = capacity
        self._head = 0
        self._tail = 0    # 指向下一个插入位置
        self._size = 0

    def enqueue(self, val):
        if self._size == self._cap:
            raise OverflowError("队列已满")
        self._data[self._tail] = val
        self._tail = (self._tail + 1) % self._cap
        self._size += 1

    def dequeue(self):
        if self._size == 0:
            raise IndexError("空队列")
        val = self._data[self._head]
        self._head = (self._head + 1) % self._cap
        self._size -= 1
        return val

# 测试
cq = CircularQueue(3)
cq.enqueue("a"); cq.enqueue("b"); cq.enqueue("c")
print(cq.dequeue())  # a
cq.enqueue("d")      # 覆盖旧位置
print(cq.dequeue(), cq.dequeue(), cq.dequeue())  # b, c, d

5.4 双端队列（Deque）

dq = deque()
dq.append("右")           # 右边进

# 经典应用：滑动窗口最大值
def max_sliding_window(nums: list[int], k: int) -> list[int]:
    """单调队列 — O(n) 解决滑动窗口最大值"""
    dq = deque()       # 存索引，单调递减
    result = []
    for i, v in enumerate(nums):
        # 移除超出窗口的左端
        if dq and dq[0] < i - k + 1:
            dq.popleft()
        # 维护单调递减
        while dq and nums[dq[-1]] <= v:
            dq.pop()
        dq.append(i)
        if i >= k - 1:
            result.append(nums[dq[0]])
    return result

print(max_sliding_window([1,3,-1,-3,5,3,6,7], 3))
# [3, 3, 5, 5, 6, 7]

🔑 6. 哈希表

哈希表（Hash Table） 通过哈希函数将 key 映射到数组索引，实现平均 O(1) 的查找/插入/删除。Python 的 dict 和 set 就是最常用的哈希表实现。

6.1 哈希表核心概念

🔢 哈希函数

key → 整数（再取模得到索引）。好的哈希函数分布均匀、冲突少

💥 哈希冲突

两个不同 key 映射到同一索引。两种解决方式：链地址法（每个槽位挂链表）和开放地址法（探测空位）

⚖ 负载因子

元素数 / 槽位数。超过阈值（如 0.75）触发扩容重哈希（rehash）

6.2 Python dict 操作

Python
d = {}

6.3 手写哈希表（链地址法）

Python
class HashTable:
    def __init__(self, capacity=8):
        self._cap = capacity
        self._size = 0
        self._buckets = [[] for _ in range(capacity)]

    def _hash(self, key):
        return hash(key) % self._cap

    def _resize(self, new_cap):
        old_buckets = self._buckets
        self._cap = new_cap
        self._size = 0
        self._buckets = [[] for _ in range(new_cap)]
        for bucket in old_buckets:
            for k, v in bucket:
                self[k] = v

    def __setitem__(self, key, val):
        idx = self._hash(key)
        bucket = self._buckets[idx]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, val)
                return
        bucket.append((key, val))
        self._size += 1
        if self._size > self._cap * 0.75:   # 负载因子 > 0.75 → 扩容
            self._resize(self._cap * 2)

    def __getitem__(self, key):
        idx = self._hash(key)
        for k, v in self._buckets[idx]:
            if k == key:
                return v
        raise KeyError(key)

    def __contains__(self, key):
        idx = self._hash(key)
        return any(k == key for k, _ in self._buckets[idx])

    def __len__(self): return self._size

    def __repr__(self):
        items = []
        for bucket in self._buckets:
            items.extend(bucket)
        return "{" + ", ".join(f"{k!r}: {v!r}" for k, v in items) + "}"

# 测试

ht["name"] = "张三"

print("age" in ht)       # True

6.4 经典哈希表题目

Python
def two_sum(nums: list[int], target: int) -> list[int]:
    """两数之和 — 经典面试题"""
    seen = {}  # {值: 索引}
    for i, v in enumerate(nums):
        complement = target - v
        if complement in seen:
            return [seen[complement], i]
        seen[v] = i
    return []

def first_non_repeating(s: str) -> str | None:
    """第一个不重复字符"""
    freq = {}
    for ch in s:
        freq[ch] = freq.get(ch, 0) + 1
    for ch in s:
        if freq[ch] == 1:
            return ch
    return None

def group_anagrams(words: list[str]) -> list[list[str]]:
    """字母异位词分组"""
    groups = {}
    for w in words:
        key = "".join(sorted(w))
        groups.setdefault(key, []).append(w)
    return list(groups.values())

print(two_sum([2, 7, 11, 15], 9))        # [0, 1]
print(first_non_repeating("abracadabra"))  # 'c'
print(group_anagrams(["eat", "tea", "tan", "ate", "nat", "bat"]))

🌳 7. 树与二叉树

树（Tree） 是非线性结构——每个节点有 0 或多个子节点，没有环。最常用的是二叉树（Binary Tree）：每个节点最多两个子节点（左子树、右子树）。

7.1 二叉树遍历

Python
class TreeNode:

        self.val = val
        self.left = left
        self.right = right

# 构建一棵树
#       1
#      / \
#     2   3
#    / \   \
#   4   5   6
root = TreeNode(1)
root.left = TreeNode(2, TreeNode(4), TreeNode(5))
root.right = TreeNode(3, None, TreeNode(6))

# 前序遍历 — 根 → 左 → 右
def preorder(node):
    if not node: return
    print(node.val, end=" ")  # 根
    preorder(node.left)          # 左
    preorder(node.right)         # 右
print("前序:", end=" "); preorder(root)  # 1 2 4 5 3 6

# 中序遍历 — 左 → 根 → 右
def inorder(node):
    if not node: return
    inorder(node.left)
    print(node.val, end=" ")
    inorder(node.right)

    """二叉树最大深度（递归）"""
    if not root: return 0

def is_balanced(root: TreeNode) -> bool:
    """判断平衡二叉树"""
    def height(node):
        if not node: return 0
        left = height(node.left)
        right = height(node.right)
        if left == -1 or right == -1 or abs(left - right) > 1:
            return -1
        return 1 + max(left, right)
    return height(root) != -1

        node.left = dfs()
        node.right = dfs()
        return node

🔍 8. 二叉搜索树 (BST)

二叉搜索树（Binary Search Tree） 满足：
左子树所有节点 < 根节点 < 右子树所有节点。
中序遍历 = 升序序列。

8.1 BST 实现

Python
class BST:
    def __init__(self):
        self.root = None

    def insert(self, val):
        """插入"""
        if not self.root:
            self.root = TreeNode(val)
            return
        cur = self.root
        while True:
            if val < cur.val:
                if not cur.left:
                    cur.left = TreeNode(val); break
                cur = cur.left
            else:
                if not cur.right:
                    cur.right = TreeNode(val); break
                cur = cur.right

    def search(self, val) -> TreeNode | None:
        """查找"""
        cur = self.root
        while cur and cur.val != val:
            cur = cur.left if val < cur.val else cur.right
        return cur

    def delete(self, val):
        """删除（递归实现）"""
        def _delete(node, val):
            if not node: return None
            if val < node.val:
                node.left = _delete(node.left, val)
            elif val > node.val:
                node.right = _delete(node.right, val)
            else:  # 找到要删除的节点
                if not node.left: return node.right
                if not node.right: return node.left
                # 有两个子节点：找右子树的最小值
                min_node = node.right
                while min_node.left: min_node = min_node.left
                node.val = min_node.val
                node.right = _delete(node.right, min_node.val)
            return node
        self.root = _delete(self.root, val)

    def to_sorted_list(self) -> list:
        """中序遍历 → 升序列表"""
        result = []
        def dfs(node):
            if not node: return
            dfs(node.left)
            result.append(node.val)
            dfs(node.right)
        dfs(self.root)
        return result

8.3 平衡二叉树 AVL 简介

BST 最坏情况（插入升序数据）退化成链表 O(n)。AVL 树通过 旋转 保证左右子树高度差 ≤ 1，所有操作 O(log n)。

四种旋转
    左左（LL）→ 右旋
    右右（RR）→ 左旋
    左右（LR）→ 左旋 + 右旋
    右左（RL）→ 右旋 + 左旋

            失衡节点
            ┌──┐
            │Z │
            └──┘
           /     \
       左重?    右重?
      (左高>右高+1)    (右高>左高+1)

  LL: Z.left 左子节点的左子树插入 → 右旋 Z
  LR: Z.left 左子节点的右子树插入 → 左旋 Z.left + 右旋 Z
💡 工程实践： 大多数场景直接用 sortedcontainers 库的 SortedList 或语言内置的有序集合，手写 AVL/红黑树只在竞赛或底层开发需要。

9.2 手写堆

Python
class MinHeap:
    def __init__(self, data=None):
        self._arr = data if data else []
        if self._arr:
            for i in range(len(self._arr)//2 - 1, -1, -1):
                self._sift_down(i)

    def _sift_down(self, i):
        n = len(self._arr)
        while True:
            smallest = i
            left = 2 * i + 1
            right = 2 * i + 2
            if left < n and self._arr[left] < self._arr[smallest]:
                smallest = left
            if right < n and self._arr[right] < self._arr[smallest]:
                smallest = right
            if smallest == i: break
            self._arr[i], self._arr[smallest] = self._arr[smallest], self._arr[i]
            i = smallest

    def _sift_up(self, i):
        while i > 0:
            parent = (i - 1) // 2
            if self._arr[parent] <= self._arr[i]: break
            self._arr[parent], self._arr[i] = self._arr[i], self._arr[parent]
            i = parent

    def push(self, val):
        self._arr.append(val)
        self._sift_up(len(self._arr) - 1)

    def pop(self):
        if not self._arr: raise IndexError
        val = self._arr[0]
        last = self._arr.pop()
        if self._arr:
            self._arr[0] = last
            self._sift_down(0)
        return val

    def peek(self): return self._arr[0]
    def __len__(self): return len(self._arr)

9.3 堆的经典应用

def top_k(nums: list[int], k: int) -> list[int]:
    """Top K 最大元素（最小堆）"""
    heap = nums[:k]
    heapq.heapify(heap)
    for v in nums[k:]:
        if v > heap[0]:
            heapq.heapreplace(heap, v)
    return heap

def median_stream(nums: list[int]) -> list[float]:
    """数据流中位数（最大堆+最小堆）"""
    left = []   # 最大堆（存负数）
    right = []  # 最小堆
    result = []

    def add(v):
        if not left or v <= -left[0]:
            heapq.heappush(left, -v)
        else:
            heapq.heappush(right, v)
        # 平衡：|left| >= |right| 且差 ≤ 1
        if len(left) > len(right) + 1:
            heapq.heappush(right, -heapq.heappop(left))
        elif len(right) > len(left):
            heapq.heappush(left, -heapq.heappop(right))

    def get_median():
        if len(left) > len(right):
            return -left[0]
        return (-left[0] + right[0]) / 2

    for v in nums:
        add(v)
        result.append(get_median())
    return result

print(top_k([3, 2, 1, 5, 6, 4], 3))             # [4, 5, 6]
print(median_stream([1, 2, 3, 4, 5]))                 # [1, 1.5, 2, 2.5, 3]

🔤 10. 字典树 (Trie)

字典树（Trie / 前缀树） 是一种多叉树，用于高效地存储和查询字符串集合。
每个节点代表一个前缀，从根到叶子的路径组成一个完整字符串。
查询"是否存在"或"以某前缀开头的所有词"——复杂度仅为 O(len(key))，与词库大小无关。

Python
class TrieNode:
    def __init__(self):
        self.children = {}   # {字符: TrieNode}
        self.is_end = False  # 是否是完整单词的结尾

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word: str):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word: str) -> bool:
        """是否完整存在"""
        node = self._find_prefix(word)
        return node is not None and node.is_end

    def starts_with(self, prefix: str) -> bool:
        """是否有以 prefix 开头的词"""
        return self._find_prefix(prefix) is not None

    def _find_prefix(self, prefix: str) -> TrieNode | None:
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return None
            node = node.children[ch]
        return node

    def autocomplete(self, prefix: str) -> list[str]:
        """前缀补全 — 返回所有匹配的完整词"""
        node = self._find_prefix(prefix)
        if not node: return []
        result = []

        def dfs(n, path):
            if n.is_end:
                result.append(prefix + path)
            for ch in sorted(n.children.keys()):
                dfs(n.children[ch], path + ch)
        dfs(node, "")
        return result

    def delete(self, word: str) -> bool:
        """删除一个词"""
        def _delete(node, word, depth):
            if depth == len(word):
                if not node.is_end: return False
                node.is_end = False
                return len(node.children) == 0  # 可删除?
            ch = word[depth]
            if ch not in node.children: return False
            should_delete = _delete(node.children[ch], word, depth + 1)
            if should_delete:
                del node.children[ch]
                return not node.is_end and len(node.children) == 0
            return False
        return _delete(self.root, word, 0)

11.4 Dijkstra 最短路径

import heapq
from collections import defaultdict

def dijkstra(graph: dict, start: str) -> dict:
    """Dijkstra 最短路径算法（优先队列优化）"""
    dist = {node: float("inf") for node in graph}
    dist[start] = 0
    pq = [(0, start)]     # (距离, 节点)
    visited = set()

    while pq:
        d, node = heapq.heappop(pq)
        if node in visited: continue
        visited.add(node)

        for neighbor, weight in graph[node].items():
            new_dist = d + weight
            if new_dist < dist[neighbor]:
                dist[neighbor] = new_dist
                heapq.heappush(pq, (new_dist, neighbor))

    return dist

from collections import defaultdict, deque

def topological_sort(edges: list[tuple[str, str]]) -> list[str]:
    """拓扑排序（Kahn 算法 — BFS 入度法）"""
    graph = defaultdict(list)
    in_degree = defaultdict(int)

    for u, v in edges:
        graph[u].append(v)
        in_degree[v] += 1
        if u not in in_degree: in_degree[u] = 0

    q = deque([node for node in in_degree if in_degree[node] == 0])
    result = []

    while q:
        node = q.popleft()
        result.append(node)
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                q.append(neighbor)

    # 如果 result 长度 < 节点数 → 有环
    return result

def kruskal_mst(nodes, edges):
    """Kruskal 最小生成树（并查集实现）"""
    parent = {n: n for n in nodes}
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]  # 路径压缩
            x = parent[x]
        return x
    def union(a, b):
        ra, rb = find(a), find(b)
        if ra == rb: return False
        parent[ra] = rb
        return True

    # 按权重排序
    sorted_edges = sorted(edges, key=lambda x: x[2])
    mst = []
    total = 0
    for u, v, w in sorted_edges:
        if union(u, v):
            mst.append((u, v, w))
            total += w
    return mst, total

    ("A", "B", 4), ("A", "C", 3),
    ("B", "C", 2), ("B", "D", 1),
    ("C", "D", 5),
]

🔗 12. 并查集

并查集（Disjoint Set Union, DSU） 用于高效地合并集合和判断元素是否在同一集合。
应用：连通分量、Kruskal 最小生成树、社交网络朋友关系。

Python
class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [1] * n       # 按秩合并优化
        self.components = n         # 连通分量数

    def find(self, x):
        """查找根（带路径压缩）"""
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]  # 隔代压缩
            x = self.parent[x]
        return x

    # 递归写法（更优雅）
    def find_r(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find_r(self.parent[x])
        return self.parent[x]

    def union(self, a, b) -> bool:
        """合并两个集合，返回是否合并成功"""
        ra, rb = self.find_r(a), self.find_r(b)
        if ra == rb: return False
        # 按秩合并：小树挂到大树上
        if self.rank[ra] < self.rank[rb]:
            ra, rb = rb, ra
        self.parent[rb] = ra
        if self.rank[ra] == self.rank[rb]:
            self.rank[ra] += 1
        self.components -= 1
        return True

    def connected(self, a, b) -> bool:
        return self.find_r(a) == self.find_r(b)

        dsu.union(a, b)

13.1 树状数组（Fenwick Tree）

class FenwickTree:
    """树状数组 — 单点更新 + 前缀和查询 O(log n)"""
    def __init__(self, n):
        self.n = n
        self.bit = [0] * (n + 1)   # 1-indexed

    def add(self, idx, delta):
        """在下标 idx 处增加 delta"""
        i = idx + 1                 # 转 1-indexed
        while i <= self.n:
            self.bit[i] += delta
            i += i & -i               # i += lowbit(i)

    def sum(self, idx):
        """前缀和 [0, idx)"""
        res = 0
        i = idx
        while i > 0:
            res += self.bit[i]
            i -= i & -i
        return res

    def range_sum(self, l, r):
        """区间和 [l, r)"""
        return self.sum(r) - self.sum(l)

13.2 线段树（Segment Tree）

Python
class SegmentTree:
    """线段树 — 区间查询 + 单点更新 O(log n)"""
    def __init__(self, data):
        self.n = len(data)
        self.tree = [0] * (4 * self.n)
        if self.n > 0:
            self._build(data, 0, 0, self.n - 1)

    def _build(self, data, node, l, r):
        if l == r:
            self.tree[node] = data[l]
            return
        mid = (l + r) // 2
        self._build(data, node * 2 + 1, l, mid)
        self._build(data, node * 2 + 2, mid + 1, r)
        self.tree[node] = self.tree[node * 2 + 1] + self.tree[node * 2 + 2]

    def update(self, idx, val):
        return self._update(0, 0, self.n - 1, idx, val)

    def _update(self, node, l, r, idx, val):
        if l == r:
            self.tree[node] = val
            return
        mid = (l + r) // 2
        if idx <= mid:
            self._update(node * 2 + 1, l, mid, idx, val)
        else:
            self._update(node * 2 + 2, mid + 1, r, idx, val)
        self.tree[node] = self.tree[node * 2 + 1] + self.tree[node * 2 + 2]

    def query(self, ql, qr):
        return self._query(0, 0, self.n - 1, ql, qr)

    def _query(self, node, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.tree[node]
        if r < ql or l > qr:
            return 0
        mid = (l + r) // 2
        return (self._query(node * 2 + 1, l, mid, ql, qr) +
                self._query(node * 2 + 2, mid + 1, r, ql, qr))

# 测试
st = SegmentTree([1, 3, 5, 7, 9, 11])
print(st.query(1, 4))   # 3+5+7+9 = 24
st.update(2, 10)        # arr[2] = 10
print(st.query(1, 4))   # 3+10+7+9 = 29
